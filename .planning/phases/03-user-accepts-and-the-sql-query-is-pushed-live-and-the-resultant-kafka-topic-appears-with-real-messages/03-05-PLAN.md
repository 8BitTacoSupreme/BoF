**Guard pattern:**
```javascript
const axios = require('axios');

const FLINK_SQL_GATEWAY_URL = process.env.FLINK_SQL_GATEWAY_URL || 'http://localhost:8083';

// Guard: only run if Flink SQL Gateway is reachable
let gatewayAvailable = false;
beforeAll(async () => {
  try {
    const res = await axios.get(`${FLINK_SQL_GATEWAY_URL}/v1/info`, { timeout: 3000 });
    gatewayAvailable = res.status === 200;
  } catch {
    gatewayAvailable = false;
  }
});

const describeIfFlinkGateway = () => gatewayAvailable ? describe : describe.skip;
// Note: since the check is async, use a different pattern:
// Use a top-level describe that checks and skips inside
```

**Better guard pattern (matches canonical.test.js approach):**
```javascript
const describeIfFlinkGateway = process.env.FLINK_GATEWAY_URL
  ? describe
  : describe.skip;

describeIfFlinkGateway('Flink Integration (requires live SQL Gateway)', () => {
  // ...tests...
}, 60000);
```

Set `FLINK_GATEWAY_URL=http://localhost:8083` when running with live Flink.

**Test cases:**

1. **Gateway health check:**
   ```javascript
   test('SQL Gateway /v1/info returns product info', async () => {
     const res = await axios.get(`${FLINK_SQL_GATEWAY_URL}/v1/info`);
     expect(res.status).toBe(200);
     expect(res.data.productName).toBe('Apache Flink');
   });
   ```

2. **Session creation:**
   ```javascript
   test('can create a SQL Gateway session', async () => {
     const res = await axios.post(`${FLINK_SQL_GATEWAY_URL}/v1/sessions`, {
       sessionName: 'integration-test',
       properties: { 'execution.runtime-mode': 'streaming' },
     });
     expect(res.status).toBe(200);
     expect(res.data.sessionHandle).toBeDefined();
     sessionHandle = res.data.sessionHandle;
   });
   ```

3. **DDL submission (simple CREATE TABLE):**
   ```javascript
   test('can submit a CREATE TABLE statement', async () => {
     const ddl = `CREATE TABLE test_source (
       id BIGINT,
       name STRING,
       event_time BIGINT,
       event_ts AS TO_TIMESTAMP_LTZ(event_time, 3),
       WATERMARK FOR event_ts AS event_ts - INTERVAL '5' SECOND
     ) WITH (
       'connector' = 'kafka',
       'topic' = 'retail.orders',
       'properties.bootstrap.servers' = 'broker:9092',
       'properties.group.id' = 'integration-test-reader',
       'scan.startup.mode' = 'earliest-offset',
       'value.format' = 'avro-confluent',
       'value.avro-confluent.url' = 'http://schema-registry:8081'
     )`;

     const res = await axios.post(
       `${FLINK_SQL_GATEWAY_URL}/v1/sessions/${sessionHandle}/statements`,
       { statement: ddl }
     );
     expect(res.status).toBe(200);
     expect(res.data.operationHandle).toBeDefined();
   });
   ```

4. **deployJob via flinkService (full flow):**
   ```javascript
   test('deployJob submits SQL and returns Running state', async () => {
     const { deployJob } = require('../src/services/flinkService');
     // Use a simple print-sink job (writes to stdout, not Kafka)
     const sql = `
       CREATE TABLE print_sink (id BIGINT, name STRING)
       WITH ('connector' = 'print');
       INSERT INTO print_sink SELECT 1 AS id, 'test' AS name FROM (VALUES (1));
     `;
     const result = await deployJob(sql);
     expect(result.operationHandle).toBeDefined();
     expect(result.sessionHandle).toBeDefined();
   }, 30000);
   ```
