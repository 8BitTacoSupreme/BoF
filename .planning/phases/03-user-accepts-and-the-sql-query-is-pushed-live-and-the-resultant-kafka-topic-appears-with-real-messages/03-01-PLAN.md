**3. Create `infrastructure/producers/lib/schemaHelper.js`:**

Shared helper that:
- Reads `.avsc` files from `/app/schemas/` (mounted via Docker volume or copied in Dockerfile)
- Actually, since producers run in Docker, read schemas inline or from environment. Simpler approach: embed the schema JSON directly in each producer file and register via `@kafkajs/confluent-schema-registry` `SchemaRegistry.register({ type: SchemaType.AVRO, schema: JSON.stringify(avroSchema) })`.
- Alternatively, load from files. The Dockerfile should `COPY ../schemas /app/schemas` — but the build context is `./producers`, so schemas are outside context. Better approach: copy schemas into producers directory at build time via docker-compose build args, OR just embed schemas directly in producer code.

**Recommendation: Embed schemas directly in each producer.** This avoids Docker build context issues and keeps each producer self-contained.

Create `lib/schemaHelper.js`:
```javascript
const { Kafka } = require('kafkajs');
const { SchemaRegistry, SchemaType } = require('@kafkajs/confluent-schema-registry');

const KAFKA_BROKERS = (process.env.KAFKA_BROKERS || 'localhost:9092').split(',');
const SCHEMA_REGISTRY_URL = process.env.SCHEMA_REGISTRY_URL || 'http://localhost:8081';

const kafka = new Kafka({ clientId: 'bof-producer', brokers: KAFKA_BROKERS });
const registry = new SchemaRegistry({ host: SCHEMA_REGISTRY_URL });

async function createTopicIfNeeded(topicName, numPartitions = 3) {
  const admin = kafka.admin();
  await admin.connect();
  const existing = await admin.listTopics();
  if (!existing.includes(topicName)) {
    await admin.createTopics({ topics: [{ topic: topicName, numPartitions, replicationFactor: 1 }] });
  }
  await admin.disconnect();
}

async function registerSchema(topicName, avroSchema) {
  const subject = `${topicName}-value`;
  const { id } = await registry.register({
    type: SchemaType.AVRO,
    schema: JSON.stringify(avroSchema),
  }, { subject });
  return id;
}

async function createProducer() {
  const producer = kafka.producer({ idempotent: true });
  await producer.connect();
  return producer;
}

async function encodeAndSend(producer, topicName, schemaId, records) {
  const messages = await Promise.all(records.map(async (record) => {
    const value = await registry.encode(schemaId, record);
    return { value };
  }));
  await producer.send({ topic: topicName, messages });
}

module.exports = { createTopicIfNeeded, registerSchema, createProducer, encodeAndSend };
```
